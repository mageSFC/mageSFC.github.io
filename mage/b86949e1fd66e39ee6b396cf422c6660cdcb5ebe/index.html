class ChangeviewtabTabContent extends Polymer.Element {
changeInfo = undefined;
instance = undefined;

static get template() {
  return Polymer.html`<div id="app">changeviewtab loading...</div><div id="style" />`;
}

async ready() {
  super.ready();

  this.changeInfo = this.content.value;

  const changeviewtab = await import("https://xxxxx/main.js");
  this.instance = changeviewtab.default(this.$.app);

  this.instance.app.provide("gerritChangeInfo", this.changeInfo);
  this.instance.app.provide("gerritPlugin", this.plugin);
  if (changeviewtab.mountStyle) {
    changeviewtab.mountStyle(this.$.style, true);
  }
  this.instance.mount();
}

}

static get template() {
return Polymer.html<gr-tooltip-content> <div style="line-height: 22px"> <img src="https://avatars.githubusercontent.com/u/16930340?s=120&v=4" style="width: 18px; height: 18px; padding: 2px; vertical-align: top"/> xxxx </div> </gr-tooltip-content>;
}
}

customElements.define(“changeviewtab-tab-content”, ChangeviewtabTabContent);
customElements.define(“changeviewtab-tab-header”, ChangeviewtabTabHeader);

Gerrit.install((plugin) => {

“change-view-tab-header”,

“change-view-tab-content”,

});

    }
}

checkConfig(element) {
    let selection = window.getSelection();
    selection.removeAllRanges();
    let range = document.createRange();
    range.selectNodeContents(element);
    selection.addRange(range);
}

generateNewPassword() {
    this.$.gitDialog.showModal();
    let json = {
        "generate": true
    };
    Promise.all([
        this.plugin.restApi().get("/config/server/info"),
        this.plugin.restApi().get("/accounts/self/username"),
        this.plugin.restApi().put("/accounts/self/password.http", json)
    ]).then(values => {
        this.username = values[1];
        let password = values[2];

        let downloadUrl = values[0].download.schemes.http.url;
        this.protocol = downloadUrl.startsWith("http:") ? "http:" : "https:";
        this.httpDomain = downloadUrl.match(/:\/\/(\S*)\/a\//)[1];

        this.setPasswordUrl(password);
        this.setCmdkeyCommand(password);
    });
}

setPasswordUrl(password) {
    this.httpPasswordUrl = this.protocol + "//" + this.username + ":" + encodeURIComponent(password) + "@" + encodeURIComponent(this.httpDomain);
}

setCmdkeyCommand(password) {
    this.httpCmdkeyCommand = "cmdkey /generic:git:" + this.protocol + "//" + this.username + "@" + this.httpDomain + " /user:" + this.username + " /pass:" + password;
}

ready() {
    super.ready();
    this.$.generate.addEventListener("click", () => this.generateNewPassword());
    this.$.closeDialog.addEventListener("click", () => {
        this.username = "";
        this.protocol = "";
        this.httpDomain = "";
        this.httpPasswordUrl = "";
        this.httpCmdkeyCommand = "";
        this.$.gitDialog.close();
    })
    this.$.linuxConfig.addEventListener("click", () => this.checkConfig(this.$.linuxConfig));
    this.$.windowsConfig.addEventListener("click", () => this.checkConfig(this.$.windowsConfig));
}

}

customElements.define(HttpPasswordScreen.is, HttpPasswordScreen);
Gerrit.install(plugin => {
plugin.registerCustomComponent(“settings-screen”, “http-password-screen”);
});
